```bash
# Download the configuration files and let Caddy create a self-signed certificate.
wget https://raw.githubusercontent.com/zitadel/zitadel/main/docs/docs/self-hosting/manage/reverseproxy/docker-compose.yaml -O docker-compose-base.yaml
wget https://raw.githubusercontent.com/zitadel/zitadel/main/docs/docs/self-hosting/manage/reverseproxy/zitadel.yaml
wget https://raw.githubusercontent.com/zitadel/zitadel/main/docs/docs/self-hosting/manage/reverseproxy/caddy/docker-compose.yaml -O docker-compose-caddy.yaml
wget https://raw.githubusercontent.com/zitadel/zitadel/main/docs/docs/self-hosting/manage/reverseproxy/caddy/caddy-external-tls.conf

# Run the database, ZITADEL and Caddy
docker compose --file docker-compose-base.yaml --file docker-compose-caddy.yaml up caddy-external-tls --detach
```
