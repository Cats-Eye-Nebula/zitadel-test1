import TabItem from "@theme/TabItem";
import Tabs from "@theme/Tabs";
import PKCE from "./authmethods/pkce.mdx";

import Basic from "./authmethods/basic.mdx";
import Implicit from "./authmethods/implicit.mdx";
import PKCENative from "./authmethods/pkcenative.mdx";
import JWTPrivateKey from "./authmethods/jwtpk.mdx";
import JWTAuthRequest from "./authrequest/jwtpk.mdx";
import BasicAuthRequest from "./authrequest/basic.mdx";
import PKCEAuthRequest from "./authrequest/pkce.mdx";
import Callback from "./callback.mdx";
import JWTTokenRequest from "./tokenrequest/jwtpk.mdx";
import PKCETokenRequest from "./tokenrequest/pkce.mdx";
import BasicTokenRequest from "./tokenrequest/basic.mdx";

export default function AuthMethods(props) {
  return props.selected == "web" ? (
    <div>
      <Tabs
        groupId="auth-type"
        default="pkce"
        values={[
          { label: "PKCE", value: "pkce" },
          { label: "Basic Auth", value: "basic" },
          { label: "JWT with Private Key", value: "jwtpk" },
        ]}
      >
        <TabItem value="pkce">
          <PKCE />
          <PKCEAuthRequest />
          <Callback />
          <PKCETokenRequest />
        </TabItem>
        <TabItem value="basic">
          <Basic />
          <BasicAuthRequest />
          <Callback />
          <BasicTokenRequest />
        </TabItem>
        <TabItem value="jwtpk">
          <JWTPrivateKey />
          <JWTAuthRequest />
          <Callback />
          <JWTTokenRequest />
        </TabItem>
      </Tabs>
    </div>
  ) : props.selected == "native" ? (
    <Tabs
      groupId="auth-type"
      default="pkce"
      values={[{ label: "PKCE", value: "pkce" }]}
    >
      <TabItem value="pkce">
        <PKCENative />
        <PKCEAuthRequest />
        <Callback />
        <PKCETokenRequest />
      </TabItem>
    </Tabs>
  ) : props.selected == "spa" ? (
    <Tabs
      groupId="auth-type"
      default="pkce"
      values={[
        { label: "PKCE", value: "pkce" },
        { label: "Implicit", value: "implicit" },
      ]}
    >
      <TabItem value="pkce">
        <PKCE />
        <PKCEAuthRequest />
        <Callback />
        <PKCETokenRequest />
      </TabItem>
      <TabItem value="implicit">
        <Implicit />
      </TabItem>
    </Tabs>
  ) : null;
}
